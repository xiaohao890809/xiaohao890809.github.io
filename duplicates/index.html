def findDuplicates(nums: List[int]) -> List[int]:
    for i in range(len(nums)):
        while nums[i] != nums[nums[i] - 1]:
            nums[nums[i] - 1], nums[i] = nums[i], nums[nums[i] - 1]
    return [num for i, num in enumerate(nums) if num - 1 != i]

def findDuplicates(nums: List[int]) -> List[int]:
    ans = []
    for x in nums:
        x = abs(x)
        if nums[x - 1] > 0:
            nums[x - 1] = -nums[x - 1]
        else:
            ans.append(x)
    return ans
